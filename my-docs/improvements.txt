3️⃣ Automatic QA checks (Python)

These catch common MT failures before human review.

import re
import unicodedata

SCRIPTS = {
    "Hindi": r"\u0900-\u097F",
    "Tamil": r"\u0B80-\u0BFF",
    "Telugu": r"\u0C00-\u0C7F",
    "Malayalam": r"\u0D00-\u0D7F",
    "Kannada": r"\u0C80-\u0CFF"
}

def qa_checks(source, translation, lang):
    issues = []

    # 1. Empty
    if not translation.strip():
        issues.append("EMPTY_OUTPUT")

    # 2. English leakage
    if re.search(r"[A-Za-z]{4,}", translation):
        issues.append("ENGLISH_WORDS_PRESENT")

    # 3. Wrong script
    if not re.search(f"[{SCRIPTS[lang]}]", translation):
        issues.append("WRONG_SCRIPT")

    # 4. Too short
    if len(translation) < 0.6 * len(source):
        issues.append("TOO_SHORT")

    # 5. Repeated sentence
    parts = translation.split("।") if lang == "Hindi" else translation.split(".")
    if len(parts) >= 3 and parts[-1] == parts[-2]:
        issues.append("DUPLICATED_SEGMENT")

    # 6. Instruction leakage
    bad_words = ["translate", "style", "output", "use formal"]
    if any(w.lower() in translation.lower() for w in bad_words):
        issues.append("PROMPT_LEAKAGE")

    return issues


Usage:

issues = qa_checks(source_text, model_output, "Malayalam")
if issues:
    print("QA FAIL:", issues)
